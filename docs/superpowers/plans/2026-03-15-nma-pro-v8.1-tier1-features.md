```javascript
const POTHCalculator={
  calculate(ranking){
    // ranking = AppState.ranking array, each element has .rankProbs (array of T probabilities)
    if(!ranking||ranking.length<2) return null;
    const T=ranking.length;
    const log2T=Math.log2(T);
    if(log2T===0) return {poth:1, entropy:0, maxEntropy:0, treatments:T};
    let totalEntropy=0;
    for(let i=0;i<T;i++){
      const probs=ranking[i].rankProbs;
      if(!probs||probs.length<T) return null;
      let Hi=0;
      for(let j=0;j<T;j++){
        const p=probs[j];
        if(p>0) Hi -= p * Math.log2(p);
      }
      totalEntropy += Hi;
    }
    const Hobs = totalEntropy / T;
    const poth = 1 - Hobs / log2T;
    return {
      poth: Math.max(0, Math.min(1, poth)),
      entropy: Hobs,
      maxEntropy: log2T,
      treatments: T,
      interpretation: poth >= 0.8 ? 'High precision' :
                      poth >= 0.5 ? 'Moderate precision' :
                      poth >= 0.2 ? 'Low precision' : 'Very low precision'
    };
  }
};
```

```javascript
const MIDAdjustedRanking={
  // Default MID by effect measure (log scale for ratio measures)
  defaults:{OR:0.5,RR:0.2,HR:0.2,SMD:0.2,RD:0.05},

  getMID(effectMeasure){
    const input=document.getElementById('midThresholdInput');
    const userVal=input?parseFloat(input.value):NaN;
    if(isFinite(userVal)&&userVal>0) return userVal;
    return this.defaults[effectMeasure]??0.5;
  },

  calculate(ranking, nmaResults, effectMeasure, nSim=2000){
    if(!ranking||ranking.length<2||!nmaResults) return null;
    const T=ranking.length;
    const mid=this.getMID(effectMeasure);
    const effects=ranking.map(r=>r.effect??0);
    const ses=ranking.map(r=>r.se??0.1);
    const lowerBetter=nmaResults.direction==='lower';

    // Seeded PRNG for reproducibility (xoshiro128**)
    let s=[0x9E3779B9,0x243F6A88,0xB7E15162,0xD1310BA6];
    function xoshiro(){let t=s[1]<<9,r;s[2]^=s[0];s[3]^=s[1];s[1]^=s[2];s[0]^=s[3];s[2]^=t;s[3]=(s[3]<<11)|(s[3]>>>21);r=s[0]+s[3];return(r>>>0)/4294967296;}
    function rnorm(){let u,v,s;do{u=2*xoshiro()-1;v=2*xoshiro()-1;s=u*u+v*v}while(s>=1||s===0);return u*Math.sqrt(-2*Math.log(s)/s);}

    // Simulate MID-adjusted ranks
    const midRankProbs=Array.from({length:T},()=>new Float64Array(T));
    const pMeaningfullyBest=new Float64Array(T);
    const midPscores=new Float64Array(T);
    const meanMidRank=new Float64Array(T);

    for(let sim=0;sim<nSim;sim++){
      // Sample treatment effects from MVN
      const sampled=effects.map((e,i)=>e+ses[i]*rnorm());
      // Build pairwise dominance with MID
      const wins=new Float64Array(T);
      for(let a=0;a<T;a++){
        for(let b=0;b<T;b++){
          if(a===b) continue;
          const diff=lowerBetter?(sampled[b]-sampled[a]):(sampled[a]-sampled[b]);
          if(diff>mid) wins[a]++;
          else if(Math.abs(diff)<=mid) wins[a]+=0.5; // tied
        }
      }
      // Convert wins to ranks (higher wins = better rank = lower number)
      const indexed=Array.from({length:T},(_,i)=>({i,w:wins[i]}));
      indexed.sort((a,b)=>b.w-a.w);
      const ranks=new Float64Array(T);
      let pos=0;
      while(pos<T){
        let end=pos+1;
        while(end<T&&indexed[end].w===indexed[pos].w) end++;
        const avgRank=(pos+end-1)/2; // 0-based average rank for ties
        for(let k=pos;k<end;k++){
          ranks[indexed[k].i]=avgRank;
          midRankProbs[indexed[k].i][Math.round(avgRank)]+=1;
        }
        pos=end;
      }
      // Track P(meaningfully best)
      const bestIdx=indexed[0].i;
      const secondBestWins=indexed.length>1?indexed[1].w:-Infinity;
      if(indexed[0].w-secondBestWins>0) pMeaningfullyBest[bestIdx]++;
      // Accumulate mean rank
      for(let i=0;i<T;i++) meanMidRank[i]+=ranks[i];
    }

    // Normalize
    for(let i=0;i<T;i++){
      for(let j=0;j<T;j++) midRankProbs[i][j]/=nSim;
      pMeaningfullyBest[i]/=nSim;
      meanMidRank[i]=meanMidRank[i]/nSim+1; // convert to 1-based
      // MID-P-score: derived from mean MID-adjusted rank
      midPscores[i]=1-((meanMidRank[i]-1)/(T-1)); // P-score from mean rank
    }

    const result=ranking.map((r,i)=>({
      treatment:r.treatment,
      originalPscore:r.pScore,
      midPscore:midPscores[i],
      pMeaningfullyBest:pMeaningfullyBest[i],
      meanMidRank:meanMidRank[i],
      midRankProbs:Array.from(midRankProbs[i])
    }));
    result.sort((a,b)=>b.midPscore-a.midPscore);
    return {mid, nSim, effectMeasure, lowerBetter, rankings:result};
  }
};
```

```javascript
const ContributionMatrix={
  description:'Evidence contribution via GLS hat matrix (Papakonstantinou et al. 2018)',
  calculate(processed, matrices, tau2, glsResult){
    if(!matrices||!matrices.X||!matrices.y) return {applicable:false,reason:'No design matrix available'};
    const {X, y, V, n, p}=matrices;
    if(n<2||p<1) return {applicable:false,reason:'Insufficient data'};

    // Get Sigma-inverse from GLS fit (handles multi-arm correlations correctly)
    let Si;
    if(glsResult&&glsResult.Si){
      Si=glsResult.Si;
    } else {
      // Fallback: construct diagonal W (two-arm only approximation)
      const Sigma=[];
      for(let i=0;i<n;i++){
        Sigma[i]=new Array(n).fill(0);
        Sigma[i][i]=V[i]+(tau2||0);
      }
      try{Si=Matrix.inverse(Sigma);}catch(e){return {applicable:false,reason:'Sigma not invertible'};}
    }

    // Hat matrix: H = X (X' Si X)^{-1} X' Si
    const Xt=Matrix.transpose(X);
    const XtSi=Matrix.multiply(Xt,Si);
    const XtSiX=Matrix.multiply(XtSi,X);
    let XtSiXinv;
    try{XtSiXinv=Matrix.inverse(XtSiX);}catch(e){return {applicable:false,reason:'X\'SiX not invertible'};}
    const A=Matrix.multiply(XtSiXinv,XtSi); // p x n
    const H=Matrix.multiply(X,A); // n x n hat matrix

    // Map studies to comparisons (edges)
    const edgeMap=new Map();
    processed.forEach((s,idx)=>{
      const key=[s.treatment1,s.treatment2].sort().join(' vs ');
      if(!edgeMap.has(key)) edgeMap.set(key,[]);
      edgeMap.get(key).push(idx);
    });

    // Get basic contrasts (columns of X correspond to treatments vs reference)
    const treatments=matrices.treatments||[...new Set(processed.flatMap(s=>[s.treatment1,s.treatment2]))].sort();
    const ref=matrices.reference||treatments[0];
    const contrastLabels=treatments.filter(t=>t!==ref).map(t=>t+' vs '+ref);

    // Aggregate hat matrix rows by edge -> contribution to each contrast
    const edges=[...edgeMap.keys()];
    const contributions=[];
    for(const edge of edges){
      const studyIndices=edgeMap.get(edge);
      const row=new Array(p).fill(0);
      for(const si of studyIndices){
        for(let j=0;j<p;j++){
          row[j]+=Math.abs(H[si][j]); // absolute contribution
        }
      }
      contributions.push({edge,values:row});
    }

    // Normalize columns to sum to 100%
    for(let j=0;j<p;j++){
      let colSum=0;
      for(const c of contributions) colSum+=c.values[j];
      if(colSum>0){
        for(const c of contributions) c.values[j]=(c.values[j]/colSum)*100;
      }
    }

    return {
      applicable:true,
      edges,
      contrastLabels,
      contributions,
      method:'GLS hat matrix (Papakonstantinou 2018)',
      hatMatrix:H
    };
  },

  renderHTML(result){
    if(!result||!result.applicable) return '<div class="alert alert--info"><div class="alert__text">'+(result?.reason||'Contribution matrix not available')+'</div></div>';
    const esc=Security.escapeHtml;
    let header='<th style="text-align:left;">Direct comparison</th>'+
      result.contrastLabels.map(l=>'<th style="text-align:right;min-width:90px;">'+esc(l)+'</th>').join('');
    let rows=result.contributions.map(c=>{
      let cells=c.values.map(v=>{
        const pct=v.toFixed(1);
        const intensity=Math.min(v/50,1);
        const bg='rgba(6,182,212,'+intensity.toFixed(2)+')';
        const color=intensity>0.5?'white':'inherit';
        return '<td style="text-align:right;font-family:var(--font-mono);background:'+bg+';color:'+color+';">'+pct+'%</td>';
      }).join('');
      return '<tr><td>'+esc(c.edge)+'</td>'+cells+'</tr>';
    }).join('');
    return '<div style="font-size:var(--text-xs);color:var(--text-tertiary);margin-bottom:var(--space-2);">Method: '+esc(result.method)+'</div>'+
      '<div class="overflow-x-auto"><table class="data-table"><thead><tr>'+header+'</tr></thead><tbody>'+rows+'</tbody></table></div>';
  }
};
```

```javascript
const HasseDiagram={
  // Build partial order from pairwise NMA results
  buildPartialOrder(pairwise, alpha=0.05){
    // pairwise: array of {t1, t2, effect, ci_lower, ci_upper, se}
    const treatments=[...new Set(pairwise.flatMap(p=>[p.t1,p.t2]))].sort();
    const T=treatments.length;
    const idx=Object.fromEntries(treatments.map((t,i)=>[t,i]));

    // Dominance matrix: dom[i][j] = true if treatment i is significantly better than j
    const dom=Array.from({length:T},()=>new Array(T).fill(false));
    for(const p of pairwise){
      const i=idx[p.t1], j=idx[p.t2];
      if(i===undefined||j===undefined) continue;
      // If CI excludes 0 (log scale) or 1 (ratio scale)
      if(p.ci_lower>0){ dom[i][j]=true; } // t1 significantly better
      else if(p.ci_upper<0){ dom[j][i]=true; } // t2 significantly better
    }

    // Transitive reduction: remove edge i->j if there exists i->k->j
    const reduced=Array.from({length:T},()=>new Array(T).fill(false));
    for(let i=0;i<T;i++) for(let j=0;j<T;j++) reduced[i][j]=dom[i][j];

    for(let i=0;i<T;i++){
      for(let j=0;j<T;j++){
        if(!reduced[i][j]) continue;
        for(let k=0;k<T;k++){
          if(k!==i&&k!==j&&dom[i][k]&&dom[k][j]){
            reduced[i][j]=false; // redundant edge
            break;
          }
        }
      }
    }

    // Assign layers (longest path from top)
    const layers=new Array(T).fill(0);
    let changed=true,maxIter=T*T;
    while(changed&&maxIter-->0){
      changed=false;
      for(let i=0;i<T;i++){
        for(let j=0;j<T;j++){
          if(dom[i][j]&&layers[j]<=layers[i]){
            layers[j]=layers[i]+1;
            changed=true;
          }
        }
      }
    }

    // Group treatments by layer
```

```javascript
const ContextualizedGRADE={
  // Classify effect magnitude relative to MID
  classifyEffect(effect, mid){
    const abs=Math.abs(effect);
    if(abs<mid) return {magnitude:'trivial', code:0};
    if(abs<2*mid) return {magnitude:effect>0?'small benefit':'small harm', code:effect>0?1:-1};
    return {magnitude:effect>0?'large benefit':'large harm', code:effect>0?2:-2};
  },

  // Map CINeMA confidence to certainty level
  certLevel(cinemaRating){
    if(!cinemaRating) return 'unknown';
    const r=cinemaRating.toLowerCase();
    if(r.includes('high')) return 'high';
    if(r.includes('moderate')) return 'moderate';
    if(r.includes('low')&&!r.includes('very')) return 'low';
    if(r.includes('very')) return 'very low';
    return 'unknown';
  },

  // Minimally contextualized: among best / intermediate / among worst
  classifyMinimal(effectClass, certainty){
    const cert=this.certLevel(certainty);
    if(cert==='very low'||cert==='unknown') return {category:'Cannot classify', css:'--text-tertiary'};
    if(effectClass.code>=2) return cert==='high'?{category:'Among the best',css:'--success-400'}:{category:'Probably among the best',css:'--success-400'};
    if(effectClass.code<=-2) return cert==='high'?{category:'Among the worst',css:'--danger-400'}:{category:'Probably among the worst',css:'--danger-400'};
    return {category:'Intermediate',css:'--warning-400'};
  },

  // Partially contextualized: 7-category scale
  classifyPartial(effect, mid, certainty){
    const cert=this.certLevel(certainty);
    const abs=Math.abs(effect);
    let category;
    if(abs<mid*0.5) category='Trivial difference';
    else if(abs<mid) category=effect>0?'Small benefit':'Small harm';
    else if(abs<2*mid) category=effect>0?'Moderate benefit':'Moderate harm';
    else category=effect>0?'Large benefit':'Large harm';

    const prefix=cert==='high'?'':cert==='moderate'?'Probably ':cert==='low'?'May have ':'Uncertain: ';
    return {
      category,
      qualified: prefix+(cert==='low'?category.toLowerCase():category),
      certainty: cert,
      magnitude: abs/mid
    };
  },

  // Run full assessment
  assess(pairwise, cinemaResults, mid, treatments, reference){
    if(!pairwise||pairwise.length===0) return null;
    const tList=treatments||[...new Set(pairwise.flatMap(p=>[p.t1,p.t2]))].sort();
    const ref=reference||tList[0];

    // Build cinema lookup
    const cinemaMap=new Map();
    if(cinemaResults&&cinemaResults.assessments){
      cinemaResults.assessments.forEach(a=>{
        cinemaMap.set(a.comparison||a.t1+' vs '+a.t2, a.overall||a.confidence||'unknown');
      });
    }

    const minimal=[], partial=[];
    for(const t of tList){
      if(t===ref) continue;
      const pw=pairwise.find(p=>(p.t1===t&&p.t2===ref)||(p.t2===t&&p.t1===ref));
      if(!pw) continue;
      const effect=pw.t1===t?pw.effect:-pw.effect;
      const key=[t,ref].sort().join(' vs ');
      const cinema=cinemaMap.get(key)||cinemaMap.get(t+' vs '+ref)||cinemaMap.get(ref+' vs '+t)||'unknown';
      const effectClass=this.classifyEffect(effect, mid);
      const minClass=this.classifyMinimal(effectClass, cinema);
      const partClass=this.classifyPartial(effect, mid, cinema);

      minimal.push({treatment:t, effect, effectClass, ...minClass, cinema});
      partial.push({treatment:t, effect, ...partClass, cinema});
    }
    return {minimal, partial, mid, reference:ref};
  }
};
```

```javascript
const ClassEffectNMA={
  // Build class mapping from data table
  buildClassMap(studies){
    const map=new Map(); // treatment -> class
    for(const s of studies){
      if(s.class1&&s.t1) map.set(s.t1, s.class1);
      if(s.class2&&s.t2) map.set(s.t2, s.class2);
    }
    return map;
  },

  // Common class effect model: constrained NMA
  analyzeCommon(processed, matrices, tau2, classMap, reference){
    const {X, y, V, n, p}=matrices;
    const treatments=processed.treatments||[...new Set(processed.flatMap(s=>[s.t1,s.t2]))].sort();
    const ref=reference||treatments[0];
    const nonRef=treatments.filter(t=>t!==ref);

    // Map treatments to class indices
    const classes=[...new Set([...classMap.values()])].sort();
    const refClass=classMap.get(ref)||ref;
    const nonRefClasses=classes.filter(c=>c!==refClass);
    if(nonRefClasses.length<1) return {applicable:false,reason:'Need at least 2 classes'};

    // Build reduced design matrix X_class: map treatment columns to class columns
    const classIdx=Object.fromEntries(nonRefClasses.map((c,i)=>[c,i]));
    const pClass=nonRefClasses.length;
    const Xclass=[];
    for(let i=0;i<n;i++){
      const row=new Array(pClass).fill(0);
      for(let j=0;j<p;j++){
        if(X[i][j]!==0){
          const trt=nonRef[j];
          const cls=classMap.get(trt)||trt;
          if(cls===refClass) continue;
          const ci=classIdx[cls];
          if(ci!==undefined) row[ci]+=X[i][j];
        }
      }
      Xclass.push(row);
    }

    // Fit GLS with class design matrix
    const classMatrices={X:Xclass, y, V, n, p:pClass};
    let glsResult;
    try{
      // Build Sigma
      const Sigma=[];
      for(let i=0;i<n;i++){
        Sigma[i]=new Array(n).fill(0);
        Sigma[i][i]=V[i]+(tau2||0);
      }
      const Si=Matrix.inverse(Sigma);
      const Xt=Matrix.transpose(Xclass);
      const XtSi=Matrix.multiply(Xt,Si);
      const XtSiX=Matrix.multiply(XtSi,Xclass);
      const XtSiy=[];
      for(let j=0;j<pClass;j++){
        let s=0;for(let i=0;i<n;i++) s+=XtSi[j][i]*y[i];
        XtSiy.push(s);
      }
      const beta=Matrix.solve(XtSiX,XtSiy);
      const vcov=Matrix.inverse(XtSiX);
      // Q statistic
      const resid=[];
      for(let i=0;i<n;i++){
        let fitted=0;for(let j=0;j<pClass;j++) fitted+=Xclass[i][j]*beta[j];
        resid.push(y[i]-fitted);
      }
      let Q=0;
      for(let i=0;i<n;i++) for(let j=0;j<n;j++) Q+=resid[i]*Si[i][j]*resid[j];

      glsResult={beta,vcov,Q,df:n-pClass,pvalue:1-Stats.pchisq(Q,n-pClass)};
    }catch(e){return {applicable:false,reason:'Class model fitting failed: '+e.message};}

    // Build class-level results
    const classResults=nonRefClasses.map((cls,i)=>({
      class:cls,
      effect:glsResult.beta[i],
      se:Math.sqrt(glsResult.vcov[i][i]),
      ci_lower:glsResult.beta[i]-getCritVal(AppState.confLevel??0.95)*Math.sqrt(glsResult.vcov[i][i]),
      ci_upper:glsResult.beta[i]+getCritVal(AppState.confLevel??0.95)*Math.sqrt(glsResult.vcov[i][i]),
      treatments:treatments.filter(t=>(classMap.get(t)||t)===cls)
    }));

    return {
      applicable:true,
      model:'common',
      classResults,
      referenceClass:refClass,
      Q:glsResult.Q,
      df:glsResult.df,
      pvalue:glsResult.pvalue,
      classes,
      nClasses:classes.length
    };
  },

  // Exchangeable class effect: meta-regression with class as covariate
  analyzeExchangeable(processed, nmaResults, classMap, reference){
    if(!nmaResults||!nmaResults.pairwise) return {applicable:false,reason:'Need NMA results first'};
    const treatments=[...new Set(nmaResults.pairwise.flatMap(p=>[p.t1,p.t2]))].sort();
    const ref=reference||treatments[0];

    // Extract treatment-level estimates vs reference
    const estimates=[];
    for(const t of treatments){
      if(t===ref) continue;
      const pw=nmaResults.pairwise.find(p=>(p.t1===t&&p.t2===ref)||(p.t2===t&&p.t1===ref));
      if(!pw) continue;
      const effect=pw.t1===t?pw.effect:-pw.effect;
      const cls=classMap.get(t)||t;
      estimates.push({treatment:t, effect, se:pw.se, class:cls});
    }

    // Class-level pooling (precision-weighted average within class)
    const classes=[...new Set(estimates.map(e=>e.class))].sort();
    const classResults=classes.map(cls=>{
      const members=estimates.filter(e=>e.class===cls);
      if(members.length===0) return null;
      let wSum=0, wEffSum=0;
      members.forEach(m=>{const w=1/(m.se*m.se);wSum+=w;wEffSum+=w*m.effect;});
      const classEffect=wEffSum/wSum;
      const classSE=Math.sqrt(1/wSum);
      const z=getCritVal(AppState.confLevel??0.95);

      // Shrinkage for individual treatments
      const tauWithin=members.length>1?
        Math.max(0, members.reduce((s,m)=>s+(m.effect-classEffect)**2,0)/(members.length-1) - 1/wSum) : 0;

      const shrunk=members.map(m=>{
        const w=tauWithin/(tauWithin+m.se*m.se);
        const shrunkEff=w*m.effect+(1-w)*classEffect;
        return {treatment:m.treatment, original:m.effect, shrunk:shrunkEff, shrinkage:1-w};
      });

      return {
        class:cls,
        effect:classEffect,
        se:classSE,
        ci_lower:classEffect-z*classSE,
        ci_upper:classEffect+z*classSE,
        tauWithin:Math.sqrt(tauWithin),
        treatments:shrunk,
        k:members.length
      };
    }).filter(Boolean);

    return {applicable:true, model:'exchangeable', classResults, referenceClass:classMap.get(ref)||ref, classes, nClasses:classes.length};
  }
};
```

```javascript
  hrToLog(hr, lower, upper){

    const logLower=Math.log(lower);

    const nmaStudies=studies.map(s=>({
      name:s.study, treatment1:s.t1, treatment2:s.t2,

    }));

    const result=FrequentistNMA.analyze(nmaStudies,{
      effectMeasure:'HR',
      estimator,
      reference:options.reference
    });
    if(!result) return {error:'NMA analysis failed'};
    result.effectMeasure='HR';
    result.isHR=true;
    return result;
  },

  // Parametric survival distributions
  survivalFunctions:{
    weibull(t, shape, scale){ return Math.exp(-Math.pow(t/scale, shape)); },
    gompertz(t, shape, scale){
      // shape can be negative (deceleration); scale must be > 0
      if(Math.abs(shape)<1e-10) return Math.exp(-scale*t); // exponential limit
      return Math.exp(-(scale/shape)*(Math.exp(shape*t)-1));
    },
    lognormal(t, mu, sigma){
      if(t<=0) return 1;
      // 1 - Phi((log(t)-mu)/sigma) using error function approximation
      const z=(Math.log(t)-mu)/sigma;
      return 1-0.5*(1+erf(z/Math.SQRT2));
    },
    loglogistic(t, shape, scale){ return 1/(1+Math.pow(t/scale, shape)); }
  },

  hazardFunctions:{
    weibull(t, shape, scale){ return (shape/scale)*Math.pow(t/scale, shape-1); },
    gompertz(t, shape, scale){ return scale*Math.exp(shape*t); },
    lognormal(t, mu, sigma){
      if(t<=0) return 0;
      const z=(Math.log(t)-mu)/sigma;
      const phi=Math.exp(-0.5*z*z)/Math.sqrt(2*Math.PI)/sigma/t;
      const S=1-0.5*(1+erf(z/Math.SQRT2));
      return S>0?phi/S:0;
    },
    loglogistic(t, shape, scale){
      const x=Math.pow(t/scale, shape);
      return (shape/scale)*Math.pow(t/scale, shape-1)/(1+x);
    }
  },

  // RMST: integral of S(t) from 0 to tMax using trapezoidal rule
  computeRMST(survFn, shape, scale, tMax, nPoints=200){
    const dt=tMax/nPoints;
    let integral=0;
    for(let i=0;i<=nPoints;i++){
      const t=i*dt;
      const St=survFn(t, shape, scale);
      integral+=St*(i===0||i===nPoints?0.5:1);
    }
    return integral*dt;
  },
```

```javascript
    error_x:{type:'data',symmetric:false,array:uppers.map((u,i)=>u-effects[i]),arrayminus:effects.map((e,i)=>e-lowers[i])},
    hovertemplate:'%{y}<br>HR: %{x:.3f}<extra></extra>'
  }],{
    title:{text:'Hazard Ratios vs '+esc(ref),font:{size:14}},
    xaxis:{title:'Hazard Ratio',type:'log',zeroline:false,
      tickvals:[0.25,0.5,0.75,1,1.5,2,3],ticktext:['0.25','0.5','0.75','1','1.5','2','3']},
    yaxis:{automargin:true},
    shapes:[{type:'line',x0:1,x1:1,y0:-0.5,y1:labels.length-0.5,line:{color:'red',dash:'dash'}}],

    paper_bgcolor:'rgba(0,0,0,0)',plot_bgcolor:'rgba(0,0,0,0)',
    font:{color:'var(--text-primary,#ccc)'}
  },{responsive:true});

  // League table
  const lt=document.getElementById('survLeagueTable');
  if(lt&&result.leagueTable){
    // Use existing league table renderer with HR formatting
    let html='<table class="league-table">';
    html+='<thead><tr><th></th>'+treatments.map(t=>'<th>'+esc(t)+'</th>').join('')+'</tr></thead><tbody>';
    treatments.forEach((t1,i)=>{
      html+='<tr><th>'+esc(t1)+'</th>';
      treatments.forEach((t2,j)=>{
        if(i===j){html+='<td class="reference-cell">'+esc(t1)+'</td>';return;}
        const pw=result.pairwise.find(p=>(p.t1===t1&&p.t2===t2)||(p.t2===t1&&p.t1===t2));
        if(!pw){html+='<td>—</td>';return;}
        const eff=pw.t1===t1?pw.effect:-pw.effect;
        const hr=Math.exp(eff).toFixed(2);
        const lo=Math.exp(eff-1.96*pw.se).toFixed(2);
        const hi=Math.exp(eff+1.96*pw.se).toFixed(2);
        const zz=getCritVal(AppState.confLevel??0.95);
        const sig=eff-zz*pw.se>0||eff+zz*pw.se<0;
        const cls=sig?(eff<0?'cell--benefit':'cell--harm'):'';
        html+='<td class="'+cls+'">'+hr+' ('+lo+', '+hi+')</td>';
      });
      html+='</tr>';
    });
    html+='</tbody></table>';
    lt.innerHTML=html;
  }

  // Ranking summary
  if(result.ranking){
    const rankDiv=document.getElementById('survRankingResults');
    if(rankDiv){
      let rows=result.ranking.map(r=>
        '<tr><td>'+esc(r.treatment)+'</td>'+
        '<td class="font-mono text-right">'+(r.pscore*100).toFixed(1)+'%</td>'+
        '<td class="font-mono text-right">'+r.meanRank.toFixed(2)+'</td></tr>'
      ).join('');
      rankDiv.innerHTML='<table class="data-table"><thead><tr><th>Treatment</th><th>P-score</th><th>Mean Rank</th></tr></thead><tbody>'+rows+'</tbody></table>';
    }
  }
}
```
