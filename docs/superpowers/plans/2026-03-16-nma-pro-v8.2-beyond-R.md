### Task 2: Forward Search Outlier Detection — ~200 lines

**Why:** NMAoutlier (Petropoulou et al. 2021) is the most principled outlier detection method in NMA. It iteratively adds studies from most to least compatible, flagging those that sharply increase Q. No other browser tool has this.

```javascript
const ForwardSearchOutlier={
  analyze(studies,options={}){
    const{effectMeasure='OR',reference,estimator='REML',alpha=0.05}=options;
    const processed=FrequentistNMA.calcEffects(studies,effectMeasure).filter(s=>Number.isFinite(s.yi)&&Number.isFinite(s.vi)&&s.vi>0);
    const n=processed.length;
    if(n<4)return{applicable:false,message:'Need at least 4 studies for forward search'};

    // Step 1: Initial fit with all studies to get residuals
    const fullResult=FrequentistNMA.analyze(studies,{effectMeasure,estimator,reference});
    if(!fullResult)return{applicable:false,message:'Full model failed'};

    // Step 2: Compute studentized residuals for each study
    const residuals=processed.map((s,i)=>{
      const fitted=this.getFittedValue(s,fullResult);
      const rawResid=s.yi-fitted;
      const studResid=rawResid/Math.sqrt(s.vi+fullResult.tau2);
      return{idx:i,study:s.name||s.treatment1+' vs '+s.treatment2,residual:rawResid,studentized:studResid,vi:s.vi};
    });

    // Step 3: Sort by absolute studentized residual (ascending = most compatible first)
    const ordered=residuals.slice().sort((a,b)=>Math.abs(a.studentized)-Math.abs(b.studentized));

    // Step 4: Forward search — add studies one at a time
    const minStart=Math.max(3,Math.ceil(n*0.5)); // Start with 50% most compatible
    const path=[];
    const outlierFlags=new Set();

    for(let k=minStart;k<=n;k++){
      const subset=ordered.slice(0,k).map(r=>studies[r.idx]);
      try{
        const subResult=FrequentistNMA.analyze(subset,{effectMeasure,estimator,reference});
        if(!subResult)continue;
        const Q=subResult.heterogeneity?.Q||0;
        const df=subResult.heterogeneity?.df||1;
        const I2=subResult.heterogeneity?.I2||0;
        const tau2=subResult.tau2||0;
        const lastAdded=ordered[k-1];

        // Q jump test: does adding this study cause disproportionate Q increase?
        const prevQ=path.length>0?path[path.length-1].Q:0;
        const deltaQ=Q-prevQ;
        const expectedDeltaQ=1; // Under null, each study adds ~1 to Q
        const qRatio=deltaQ/Math.max(0.01,expectedDeltaQ);

        path.push({
          k,study:lastAdded.study,Q,df,I2,tau2,
          deltaQ,qRatio,
          studentized:lastAdded.studentized,
          isOutlier:qRatio>Stats.qchisq?Stats.qchisq(1-alpha,1):3.84 // chi2(1) critical value
        });

        if(path[path.length-1].isOutlier)outlierFlags.add(lastAdded.study);
      }catch(e){continue;}
    }

    // Step 5: Also flag by Cook's D analog
    const cooksD=residuals.map(r=>{
      const h=1/n; // Simplified hat value
      const d=(r.studentized**2*h)/((1-h)*fullResult.matrices?.p||1);
      return{...r,cooksD:d};
    });
    const cooksCutoff=4/n;
    cooksD.filter(r=>r.cooksD>cooksCutoff).forEach(r=>outlierFlags.add(r.study));

    return{
      applicable:true,
      path,
      residuals:cooksD.sort((a,b)=>Math.abs(b.studentized)-Math.abs(a.studentized)),
      outliers:[...outlierFlags],
      nStudies:n,
      nOutliers:outlierFlags.size,
      cutoffs:{alpha,cooksCutoff},
      interpretation:outlierFlags.size===0
        ?'No outlying studies detected by forward search.'
        :`${outlierFlags.size} potential outlier(s): ${[...outlierFlags].join(', ')}. Consider sensitivity analysis excluding these studies.`,
      citation:'Petropoulou et al., Stat Methods Med Res 2021 (NMAoutlier)'
    };
  },

  getFittedValue(study,result){
    if(!result.effects?.effects)return 0;
    const eff1=result.effects.effects[study.treatment1];
    const eff2=result.effects.effects[study.treatment2];
    return(eff1?.estimate||0)-(eff2?.estimate||0);
  }
};
```

```javascript
const LeverageResidualDiag={
  analyze(results){
    if(!results||!results.matrices||!results.processed)return{applicable:false,message:'Run analysis first'};
    const{X,y,v,n,p}=results.matrices;
    const tau2=results.tau2||0;
    const treatments=results.matrices.treatments;

    // Build weight matrix W = diag(1/(v_i + tau2))
    const w=[];
    for(let i=0;i<n;i++)w.push(1/(v[i]+tau2));

    // Hat matrix H = W^{1/2} X (X'WX)^{-1} X' W^{1/2}
    // Simplified: h_ii = diagonal of X(X'WX)^{-1}X'W
    const XtWX=Matrix.zeros(p,p);
    for(let i=0;i<n;i++)for(let j=0;j<p;j++)for(let k=0;k<p;k++)
      XtWX[j][k]+=X[i][j]*w[i]*X[i][k];

    let XtWXinv;
    try{XtWXinv=Matrix.inverse(XtWX);}catch(e){return{applicable:false,message:'Singular hat matrix'};}

    // Compute hat values and residuals
    const beta=results.effects?.beta||[];
    const diagnostics=[];

    for(let i=0;i<n;i++){
      // Fitted value
      let fitted=0;
      for(let j=0;j<p;j++)fitted+=X[i][j]*beta[j];

      // Raw residual
      const rawResid=y[i]-fitted;

      // Hat value h_ii
      let h=0;
      for(let j=0;j<p;j++)for(let k=0;k<p;k++)
        h+=X[i][j]*XtWXinv[j][k]*X[i][k]*w[i];
      h=Math.min(0.999,Math.max(0,h));

      // Studentized residual
      const se=Math.sqrt((v[i]+tau2)*(1-h));
      const studentized=se>1e-10?rawResid/se:0;

      // Cook's distance
      const cooksD=(studentized**2*h)/((1-h)*p);

      // Deviance residual (signed contribution to Q)
      const devResid=Math.sign(rawResid)*Math.sqrt(w[i]*rawResid**2);

      const study=results.processed[i];
      diagnostics.push({
        study:study?.name||study?.treatment1+' vs '+study?.treatment2||'Study '+(i+1),
        comparison:(study?.treatment1||'?')+' vs '+(study?.treatment2||'?'),
        rawResid,fitted,
        hat:h,studentized,cooksD,devResid,
        weight:w[i],variance:v[i]
      });
    }

    // Thresholds
    const cooksCutoff=4/n;
    const hatCutoff=2*p/n;
    const outliers=diagnostics.filter(d=>Math.abs(d.studentized)>2||d.cooksD>cooksCutoff);

    return{
      applicable:true,
      diagnostics,
      outliers,
      thresholds:{cooksCutoff,hatCutoff,studentizedCutoff:2},
      summary:{
        nStudies:n,nOutliers:outliers.length,
        maxCooksD:Math.max(...diagnostics.map(d=>d.cooksD)),
        maxStudentized:Math.max(...diagnostics.map(d=>Math.abs(d.studentized))),
        meanHat:diagnostics.reduce((s,d)=>s+d.hat,0)/n
      },
      interpretation:outliers.length===0
        ?'No influential studies detected. Model fit is adequate.'
        :`${outliers.length} influential study/studies detected: ${outliers.map(o=>o.study).join(', ')}.`,
      citation:'Viechtbauer & Cheung, Res Synth Methods 2010; Noma et al. 2020'
    };
  }
};
```

### Task 4: Arm-Level Binomial Likelihood NMA — ~250 lines

**Why:** multinma and bnma can fit arm-level binomial models directly, avoiding the normal approximation that fails for rare events. This is THE critical gap for oncology/safety NMAs.

```javascript
const ArmLevelBinomialNMA={
  analyze(studies,options={}){
    const{reference,nIter=2000,burnin=500}=options;
    // Filter to studies with binary data
    const valid=studies.filter(s=>Number.isFinite(s.events1)&&Number.isFinite(s.n1)&&Number.isFinite(s.events2)&&Number.isFinite(s.n2)&&s.n1>0&&s.n2>0);
    if(valid.length<3)return{applicable:false,message:'Need at least 3 studies with binary event data (E1, N1, E2, N2)'};

    const tSet=new Set();
    valid.forEach(s=>{tSet.add(s.treatment1);tSet.add(s.treatment2)});
    const treatments=[...tSet].sort();
    const ref=reference||treatments[0];
    const refIdx=treatments.indexOf(ref);
    if(refIdx>0){treatments.splice(refIdx,1);treatments.unshift(ref);}
    const T=treatments.length;

    // Arm-level binomial likelihood with logit link
    // Model: events_ik ~ Binomial(n_ik, p_ik)
    //        logit(p_ik) = mu_i + delta_ik (delta_i,ref = 0)
    //        delta_ik ~ Normal(d_k - d_ref, tau^2)
    // d_ref = 0 (reference)

    // MCMC via Metropolis-Hastings
    const nStudies=valid.length;
    const d=new Float64Array(T); // treatment effects (d[0]=0 fixed)
    let tau2=0.1;
    const mu=new Float64Array(nStudies); // study baselines

    // Initialize mu from observed log-odds
    valid.forEach((s,i)=>{
      const p1=Math.max(0.01,Math.min(0.99,(s.events1+0.5)/(s.n1+1)));
      mu[i]=Math.log(p1/(1-p1));
    });

    // Log-likelihood for one study arm
    function logBinom(events,n,logitP){
      const p=1/(1+Math.exp(-logitP));
      if(p<=0||p>=1)return-1e10;
      return events*Math.log(p)+(n-events)*Math.log(1-p);
    }

    // Full log-posterior
    function logPosterior(){
      let lp=0;
      // Likelihood
      valid.forEach((s,i)=>{
        const t1idx=treatments.indexOf(s.treatment1);
        const t2idx=treatments.indexOf(s.treatment2);
        lp+=logBinom(s.events1,s.n1,mu[i]+d[t1idx]);
        lp+=logBinom(s.events2,s.n2,mu[i]+d[t2idx]);
      });
      // Prior on d: Normal(0,10^2) — vague
      for(let k=1;k<T;k++)lp-=d[k]*d[k]/200;
      // Prior on tau2: half-normal(0,1)
      if(tau2<0)return-1e10;
      lp-=tau2/2;
      // Prior on mu: Normal(0,10^2)
      for(let i=0;i<nStudies;i++)lp-=mu[i]*mu[i]/200;
      return lp;
    }

    // Metropolis-Hastings
    const samples={d:Array.from({length:T},()=>[]),tau2:[],mu:Array.from({length:nStudies},()=>[])};
    let accepted=0,total=0;
    const stepSize={d:0.15,mu:0.2,tau2:0.05};

    function rnorm(){let u=0,v=0;while(u===0)u=Math.random();while(v===0)v=Math.random();return Math.sqrt(-2*Math.log(u))*Math.cos(2*Math.PI*v);}

    for(let iter=0;iter<nIter+burnin;iter++){
      const oldLP=logPosterior();

      // Propose d (skip d[0]=0)
      for(let k=1;k<T;k++){
        const old=d[k];
        d[k]+=rnorm()*stepSize.d;
        const newLP=logPosterior();
        if(Math.log(Math.random())<newLP-oldLP){accepted++;} else{d[k]=old;}
        total++;
      }

      // Propose mu
      for(let i=0;i<nStudies;i++){
        const old=mu[i];
        mu[i]+=rnorm()*stepSize.mu;
        const newLP=logPosterior();
        if(Math.log(Math.random())<newLP-logPosterior()+newLP-logPosterior()){}
        // Simplified: accept/reject
        const lp2=logPosterior();
        if(lp2<oldLP&&Math.log(Math.random())>lp2-oldLP){mu[i]=old;}
      }

      // Propose tau2
      const oldTau=tau2;
      tau2=Math.abs(tau2+rnorm()*stepSize.tau2);
      if(logPosterior()<oldLP&&Math.log(Math.random())>logPosterior()-oldLP){tau2=oldTau;}

      // Store post-burnin
      if(iter>=burnin){
        for(let k=0;k<T;k++)samples.d[k].push(d[k]);
        samples.tau2.push(tau2);
      }
    }

    // Summarize posteriors
    const effects={};
    for(let k=0;k<T;k++){
      const s=samples.d[k].sort((a,b)=>a-b);
      const mean=s.reduce((a,b)=>a+b,0)/s.length;
      const median=s[Math.floor(s.length/2)];
      const ci=[s[Math.floor(s.length*0.025)],s[Math.floor(s.length*0.975)]];
      const or=Math.exp(mean);
      const orCI=[Math.exp(ci[0]),Math.exp(ci[1])];
      effects[treatments[k]]={mean,median,ci,or,orCI,sd:Math.sqrt(s.reduce((a,b)=>a+(b-mean)**2,0)/(s.length-1))};
    }

    const tau2Samples=samples.tau2.sort((a,b)=>a-b);
    const tau2Summary={
      mean:tau2Samples.reduce((a,b)=>a+b,0)/tau2Samples.length,
      median:tau2Samples[Math.floor(tau2Samples.length/2)],
      ci:[tau2Samples[Math.floor(tau2Samples.length*0.025)],tau2Samples[Math.floor(tau2Samples.length*0.975)]]
    };

    return{
      applicable:true,
      effects,treatments,reference:ref,
      tau2:tau2Summary,
      nStudies:valid.length,nTreatments:T,
      mcmc:{nIter,burnin,acceptRate:accepted/total},
      interpretation:`Arm-level binomial NMA with ${T} treatments and ${valid.length} studies. Posterior median tau = ${Math.sqrt(tau2Summary.median).toFixed(3)}.`,
      citation:'Dias et al., Stat Med 2013; Phillippo et al. (multinma) 2020'
    };
  }
};
```

```javascript
const MantelHaenszelNMA={
  analyze(studies,options={}){
    const{reference,effectMeasure='OR'}=options;
    if(!['OR','RR','RD'].includes(effectMeasure))return{applicable:false,message:'MH-NMA requires binary data (OR/RR/RD)'};

    const valid=studies.filter(s=>Number.isFinite(s.events1)&&Number.isFinite(s.n1)&&Number.isFinite(s.events2)&&Number.isFinite(s.n2)&&s.n1>0&&s.n2>0);
    if(valid.length<3)return{applicable:false,message:'Need at least 3 studies with binary data'};

    const tSet=new Set();valid.forEach(s=>{tSet.add(s.treatment1);tSet.add(s.treatment2)});
    const treatments=[...tSet].sort();
    const ref=reference||treatments[0];
    const refIdx=treatments.indexOf(ref);
    if(refIdx>0){treatments.splice(refIdx,1);treatments.unshift(ref);}

    // For each treatment pair with direct evidence, compute MH estimate
    const pairwise=new Map();
    for(const t of treatments){
      if(t===ref)continue;
      // Find studies comparing t vs ref (directly)
      const direct=valid.filter(s=>(s.treatment1===t&&s.treatment2===ref)||(s.treatment2===t&&s.treatment1===ref));
      if(direct.length===0)continue;

      let mhNum=0,mhDen=0,varSum=0;
      direct.forEach(s=>{
        const swap=s.treatment2===t;
        const a=swap?s.events2:s.events1; // events in treatment
        const b=swap?(s.n2-s.events2):(s.n1-s.events1);
        const c=swap?s.events1:s.events2; // events in reference
        const d=swap?(s.n1-s.events1):(s.n2-s.events2);
        const N=a+b+c+d;

        if(effectMeasure==='OR'){
          mhNum+=a*d/N;
          mhDen+=b*c/N;
          // Robins-Breslow-Greenland variance
          const P=(a+d)/N,Q=(b+c)/N;
          varSum+=(P*a*d)/(N*N*mhNum*mhNum||1)+(P*b*c+Q*a*d)/(2*N*N*mhNum*mhDen||1)+(Q*b*c)/(N*N*mhDen*mhDen||1);
        }else if(effectMeasure==='RR'){
          mhNum+=a*(c+d)/N;
          mhDen+=c*(a+b)/N;
        }else{// RD
          mhNum+=(a*(c+d)-c*(a+b))/N;
          mhDen+=(a+b)*(c+d)/N;
        }
      });

      if(mhDen===0)continue;
      const estimate=effectMeasure==='OR'?Math.log(mhNum/mhDen):effectMeasure==='RR'?Math.log(mhNum/mhDen):mhNum/mhDen;
      const isLog=effectMeasure!=='RD';
      const se=isLog?Math.sqrt(varSum||1/(mhNum+0.5)):Math.sqrt(1/mhDen);
      const z=getCritVal(AppState.confLevel??0.95);

      pairwise.set(t,{
        treatment:t,estimate,se,
        ci_lower:estimate-z*se,ci_upper:estimate+z*se,
        display:isLog?Math.exp(estimate):estimate,
        displayCI:isLog?[Math.exp(estimate-z*se),Math.exp(estimate+z*se)]:[estimate-z*se,estimate+z*se],
        nStudies:direct.length,
        hasZeroCells:direct.some(s=>s.events1===0||s.events2===0||(s.n1-s.events1)===0||(s.n2-s.events2)===0)
      });
    }

    // Also compute standard NMA for comparison
    const nmaResult=FrequentistNMA.analyze(valid,{effectMeasure,reference:ref,estimator:'DL'});

    const mhResults=[...pairwise.values()].sort((a,b)=>a.estimate-b.estimate);
    const zeroCellCount=mhResults.filter(r=>r.hasZeroCells).length;

    return{
      applicable:true,
      mhResults,treatments,reference:ref,
      nmaComparison:nmaResult,
      nStudies:valid.length,
      zeroCellStudies:zeroCellCount,
      interpretation:zeroCellCount>0
        ?`MH-NMA avoids continuity corrections for ${zeroCellCount} comparison(s) with zero cells. Compare with standard NMA to assess approximation impact.`
        :'No zero-cell studies. MH and standard NMA should agree closely.',
      citation:'Mantel & Haenszel 1959; Stijnen et al., Stat Med 2010; Efthimiou et al. 2024'
    };
  }
};
```

```javascript
const RankingStabilitySurface={
  analyze(studies,ranking,options={}){
    const{effectMeasure='OR',direction='lower',nTauPoints=8,reference}=options;
    if(!ranking||ranking.length<3)return{applicable:false,message:'Need at least 3 treatments'};

    const treatments=ranking.map(r=>r.treatment);

    const tauRange=[];
    for(let i=0;i<=nTauPoints;i++)tauRange.push(baseTau2*i*2/nTauPoints);

    // For each tau2 value, recompute rankings
    const surface=[]; // [{tau2, treatment, pScore, rank}]
    const topTreatmentByTau=[];

    for(const tau2Val of tauRange){

      const effects=FrequentistNMA.calcTreatmentEffects(result.matrices,tau2Val,'hksj');

        surface.push({tau2:tau2Val,treatment:r.treatment,pScore:r.pScore,rank:rank+1});
      });
      topTreatmentByTau.push({tau2:tau2Val,best:ranked[0].treatment,pScore:ranked[0].pScore});

    const topChanges=topTreatmentByTau.filter((t,i)=>i>0&&t.best!==topTreatmentByTau[i-1].best).length;
    const isStable=topChanges===0;

    return{

      surface,tauRange,treatments,
      topTreatmentByTau,topChanges,isStable,
      baseTau2,
      interpretation:isStable
        ?`Rankings are stable: ${topTreatmentByTau[0].best} remains best across all tau² values.`
        :`Rankings are SENSITIVE to heterogeneity: top treatment changes ${topChanges} time(s) across tau² range. ${topTreatmentByTau.map(t=>t.best).filter((v,i,a)=>a.indexOf(v)===i).join(', ')} each hold the top position.`,
```

```javascript
    const covariateTests=[];

      });
    }

    const maleGroups=[...comparisons.values()].map(c=>c.propMales).filter(m=>m.length>0);
    if(maleGroups.length>=2){
      const{H,p}=this.kruskalWallis(maleGroups);
      covariateTests.push({covariate:'% Male',test:'Kruskal-Wallis',statistic:H,pValue:p,
        balanced:p>0.1,means:maleGroups.map((g,i)=>({comparison:[...comparisons.keys()][i],mean:g.reduce((a,b)=>a+b,0)/g.length,n:g.length}))
      });
    }

    // Risk of Bias (chi-squared)
    const robGroups=[...comparisons.values()].map(c=>c.robs).filter(r=>r.length>0);
    if(robGroups.length>=2){
      const categories=['low','some','high'];
      const observed=robGroups.map(g=>categories.map(c=>g.filter(r=>r===c).length));
      const{chi2,p}=this.chiSquared(observed);
      covariateTests.push({covariate:'Risk of Bias',test:'Chi-squared',statistic:chi2,pValue:p,balanced:p>0.1});

      citation:'Salanti, Res Synth Methods 2012; Cipriani et al., Ann Intern Med 2013'

  kruskalWallis(groups){
    // Simplified Kruskal-Wallis H test
    const all=[];
    groups.forEach((g,gi)=>g.forEach(v=>all.push({v,g:gi})));
    all.sort((a,b)=>a.v-b.v);
    all.forEach((item,i)=>item.rank=i+1);
    const N=all.length;
    const k=groups.length;
    let H=0;
    groups.forEach((g,gi)=>{
      const ranks=all.filter(a=>a.g===gi).map(a=>a.rank);
      const Ri=ranks.reduce((s,r)=>s+r,0);
      const ni=ranks.length;
      if(ni>0)H+=Ri*Ri/ni;
    });
    H=(12/(N*(N+1)))*H-3*(N+1);
    const df=k-1;
    const p=df>0?1-Stats.pchisq(Math.max(0,H),df):1;
    return{H,p,df};
  },

  chiSquared(observed){
    const nRows=observed.length;
    const nCols=observed[0]?.length||0;
    if(nRows<2||nCols<2)return{chi2:0,p:1};
    const rowTotals=observed.map(r=>r.reduce((s,v)=>s+v,0));
    const colTotals=Array(nCols).fill(0);
    observed.forEach(r=>r.forEach((v,j)=>colTotals[j]+=v));
    const N=rowTotals.reduce((s,v)=>s+v,0);
    if(N===0)return{chi2:0,p:1};
    let chi2=0;
    observed.forEach((r,i)=>r.forEach((o,j)=>{
      const e=rowTotals[i]*colTotals[j]/N;
      if(e>0)chi2+=(o-e)**2/e;
    }));
    const df=(nRows-1)*(nCols-1);
    return{chi2,p:df>0?1-Stats.pchisq(chi2,df):1};
  }
};
```

### Task 9: CINeMA-Weighted Rankings — ~120 lines

**Why:** Standard rankings ignore certainty of evidence. A treatment might be "best" only through very-low-certainty evidence. This novel method down-weights comparisons with low CINeMA certainty. No R package has this.

```javascript
const CINeMAWeightedRanking={
  analyze(ranking,cinemaResults,results,options={}){
    if(!ranking||!results)return{applicable:false,message:'Need ranking and results'};
    const direction=options.direction||AppState.direction||'lower';
    const treatments=results.treatments||[];
    const ref=results.reference||treatments[0];

    // Build certainty weights from CINeMA
    const certWeights={'high':1.0,'moderate':0.75,'low':0.5,'very low':0.25,'unknown':0.5};
    const pairwise=typeof buildPairwiseFromResults==='function'?buildPairwiseFromResults(results):[];

    // Map certainty to each pairwise comparison
    const certMap=new Map();
    if(cinemaResults?.assessments){
      cinemaResults.assessments.forEach(a=>{
        const key=(a.comparison||a.t1+' vs '+a.t2).toLowerCase();
        certMap.set(key,(a.overall||a.confidence||'unknown').toLowerCase());
      });
    }

    // Recompute weighted P-scores: down-weight comparisons with low certainty
    const pScores={};
    for(let i=0;i<treatments.length;i++){
      const ti=treatments[i];
      let sumP=0,sumW=0;
      for(let j=0;j<treatments.length;j++){
        if(i===j)continue;
        const tj=treatments[j];
        const pw=pairwise.find(p=>(p.t1===ti&&p.t2===tj)||(p.t1===tj&&p.t2===ti));
        if(!pw)continue;
        const diff=pw.t1===ti?pw.effect:-pw.effect;
        const se=pw.se||0.5;
        const z=diff/se;
        const p=direction==='lower'?Stats.pnorm(-z):Stats.pnorm(z);

        // Get certainty weight
        const key1=(ti+' vs '+tj).toLowerCase();
        const key2=(tj+' vs '+ti).toLowerCase();
        const cert=certMap.get(key1)||certMap.get(key2)||'unknown';
        const w=certWeights[cert]||0.5;

        sumP+=p*w;
        sumW+=w;
      }
      pScores[ti]=sumW>0?sumP/sumW:0.5;
    }

    // Build weighted ranking
    const weightedRanking=treatments.map(t=>({
      treatment:t,
      pScore:pScores[t],
      originalPScore:ranking.find(r=>r.treatment===t)?.pScore||0,
      delta:pScores[t]-(ranking.find(r=>r.treatment===t)?.pScore||0)
    })).sort((a,b)=>b.pScore-a.pScore).map((item,i)=>({...item,rank:i+1,
      originalRank:ranking.findIndex(r=>r.treatment===item.treatment)+1
    }));

    // Identify rank changes
    const rankChanges=weightedRanking.filter(r=>r.rank!==r.originalRank);

    return{
      applicable:true,
      weightedRanking,rankChanges,
      hasCINeMA:cinemaResults?.assessments?.length>0,
      interpretation:rankChanges.length===0
        ?'Rankings unchanged after certainty weighting — all comparisons have similar certainty.'
        :`${rankChanges.length} treatment(s) changed rank after CINeMA weighting. ${weightedRanking[0].treatment} is the certainty-weighted best treatment.`,
      citation:'Novel: extends Nikolakopoulou et al. 2020 CINeMA with certainty-weighted P-scores'
    };
  }
};
```

### Task 10: Prediction Ranking Intervals — ~150 lines

**Why:** Current rank CIs are based on sampling uncertainty only. Prediction intervals account for between-study heterogeneity — where will a treatment RANK in a future study? Novel frequentist extension.

```javascript
const PredictionRankingInterval={
  analyze(ranking,results,options={}){
    if(!ranking||!results)return{applicable:false,message:'Need ranking and results'};
    const{nSim=5000,direction}=options;
    const dir=direction||AppState.direction||'lower';
    const treatments=results.treatments||[];
    const T=treatments.length;
    const tau2=results.tau2||0;
    const effects=results.effects;
    if(!effects||T<3)return{applicable:false,message:'Need at least 3 treatments'};

    // Simulate future study effects (prediction distribution)
    // For each treatment k: d_k^new ~ Normal(d_k, se_k^2 + tau^2)
    const seed=ranking.seed||42;
    const rng=TreatmentRanking.createSeededRNG(seed+999);

    const predRankCounts=treatments.map(()=>Array(T).fill(0));

    for(let sim=0;sim<nSim;sim++){
      // Sample from prediction distribution
      const simEffects=treatments.map((t,i)=>{
        if(i===0)return 0; // reference
        const est=effects.effects?.[t]?.estimate||0;
        const se=effects.effects?.[t]?.se||0.5;
        const predSE=Math.sqrt(se*se+tau2);
        return est+rng.rnorm(0,predSE);
      });

      // Rank
      const indices=treatments.map((_,i)=>i);
      dir==='lower'?indices.sort((a,b)=>simEffects[a]-simEffects[b]):indices.sort((a,b)=>simEffects[b]-simEffects[a]);
      indices.forEach((tIdx,rank)=>predRankCounts[tIdx][rank]++);
    }

    // Summarize
    const predRanking=treatments.map((t,i)=>{
      const probs=predRankCounts[i].map(c=>c/nSim);
      const meanRank=probs.reduce((s,p,r)=>s+p*(r+1),0);
      const varRank=probs.reduce((s,p,r)=>s+p*(r+1-meanRank)**2,0);
      const sorted=[];
      let cum=0;
      for(let r=0;r<T;r++){cum+=probs[r];sorted.push({rank:r+1,cumProb:cum});}
      const piLower=sorted.find(s=>s.cumProb>=0.025)?.rank||1;
      const piUpper=sorted.reverse().find(s=>s.cumProb<=0.975)?.rank||T;

      return{
        treatment:t,
        meanPredRank:meanRank,
        predRankPI:[piLower,T+1-sorted.find(s=>(1-s.cumProb)>=0.025)?.rank||T],
        predRankProbs:probs,
        originalMeanRank:ranking.find(r=>r.treatment===t)?.meanRank||0,
        widening:Math.sqrt(varRank)-Math.sqrt((ranking.find(r=>r.treatment===t)?.rankCI||[0,0]).reduce((s,v,i)=>s+(i===0?-v:v),0)**2/12||0)
      };
    }).sort((a,b)=>a.meanPredRank-b.meanPredRank);

    return{
      applicable:true,
      predRanking,nSim,tau2,
      interpretation:`Prediction ranking intervals account for tau²=${tau2.toFixed(4)}. ${predRanking[0].treatment} has the best predicted rank (mean ${predRanking[0].meanPredRank.toFixed(1)}). Wider intervals indicate less predictable rankings in future studies.`,
      citation:'Novel: prediction intervals for treatment ranks under heterogeneity (extends Nikolakopoulou et al. 2023)'
    };
  }
};
```

### Task 11: Treatment Effect Half-Life — ~150 lines

**Why:** In living NMA, evidence ages. This novel method fits exponential decay to effect sizes over time, estimating when the treatment effect degrades by 50%. Flags stale evidence. No tool does this.

- [ ] **Step 1: Insert TreatmentHalfLife class**

```javascript
const TreatmentHalfLife={
  analyze(studies,results,options={}){
    const{effectMeasure='OR'}=options;
    const processed=FrequentistNMA.calcEffects(studies,effectMeasure).filter(s=>Number.isFinite(s.yi)&&Number.isFinite(s.vi)&&s.vi>0&&Number.isFinite(s.year));
    if(processed.length<5)return{applicable:false,message:'Need at least 5 studies with year data'};

    const treatments=results?.treatments||[...new Set(processed.flatMap(s=>[s.treatment1,s.treatment2]))].sort();
    const ref=results?.reference||treatments[0];

    // For each non-reference treatment, fit temporal decay model
    const decayResults=[];

    for(const trt of treatments){
      if(trt===ref)continue;
      // Find studies involving this treatment
      const trtStudies=processed.filter(s=>s.treatment1===trt||s.treatment2===trt);
      if(trtStudies.length<3)continue;

      // Extract effect vs year (adjust sign for direction)
      const points=trtStudies.map(s=>{
        const eff=s.treatment1===trt?s.yi:-s.yi;
        return{year:s.year,effect:Math.abs(eff),se:Math.sqrt(s.vi),weight:1/s.vi};
      }).sort((a,b)=>a.year-b.year);

      const minYear=points[0].year;
      const maxYear=points[points.length-1].year;
      const span=maxYear-minYear;
      if(span<2)continue;

      // Weighted linear regression of |effect| on year
      const xMean=points.reduce((s,p)=>s+p.weight*p.year,0)/points.reduce((s,p)=>s+p.weight,0);
      const yMean=points.reduce((s,p)=>s+p.weight*p.effect,0)/points.reduce((s,p)=>s+p.weight,0);

      let num=0,den=0;
      points.forEach(p=>{num+=p.weight*(p.year-xMean)*(p.effect-yMean);den+=p.weight*(p.year-xMean)**2;});
      const slope=den>0?num/den:0;
      const intercept=yMean-slope*xMean;

      // Half-life: time for effect to decay by 50%
      // If slope < 0 (effect diminishing over time), halfLife = -0.5*intercept/slope years from first study
      const halfLife=slope<0&&intercept>0?(-0.5*intercept/slope)-minYear:null;

      // R² for temporal trend
      const ssRes=points.reduce((s,p)=>s+p.weight*(p.effect-(intercept+slope*p.year))**2,0);
      const ssTot=points.reduce((s,p)=>s+p.weight*(p.effect-yMean)**2,0);
      const R2=ssTot>0?1-ssRes/ssTot:0;

      // Slope p-value
      const slopeSE=den>0?Math.sqrt(ssRes/((points.length-2)*den)):Infinity;
      const tStat=slopeSE>0?slope/slopeSE:0;
      const pValue=2*(1-Stats.pnorm(Math.abs(tStat)));

      decayResults.push({
        treatment:trt,slope,intercept,R2,pValue,
        halfLife,nStudies:points.length,yearSpan:span,
        trend:slope<-0.001?'declining':slope>0.001?'increasing':'stable',
        points
      });
    }

    const declining=decayResults.filter(d=>d.trend==='declining'&&d.pValue<0.1);

    return{
      applicable:true,
      decayResults:decayResults.sort((a,b)=>a.slope-b.slope),
      declining,
      interpretation:declining.length===0
        ?'No significant temporal decline detected for any treatment.'
        :`${declining.length} treatment(s) show declining effects over time: ${declining.map(d=>d.treatment+(d.halfLife?` (half-life ~${d.halfLife.toFixed(0)} years)`:'')).join(', ')}.`,
      citation:'Novel: treatment effect temporal decay analysis for living NMA'
    };
  }
};
```
